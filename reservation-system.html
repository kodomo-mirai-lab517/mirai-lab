// ============================================================ // かめや 給料自動計算スクリプト v5 // 新フォーム（2026年6月〜）専用 // ・flat staff: 給料1列 // ・tiered staff: 給料を〜22時／22時〜の2列に分割 // ============================================================ const CONFIG = { FORM_SHEET: 'フォームの回答 1', STAFF: { '小笠原祐美': { nick: 'ゆみさん', type: 'flat', rate: 1600 }, '高橋由美子': { nick: 'たかはし', type: 'tiered' }, '板橋香津枝': { nick: 'かずえ', type: 'flat', rate: 1500 }, '森蔭佐知': { nick: 'さち', type: 'tiered' }, }, TIERED: { before22: 1300, after22: 1500 }, ALLOWANCE: { douhan: 1000, champagne: 1000, insta: 500, raitenRate: 0.15 }, }; const COL = { timestamp: 0, email: 1, name: 2, workDate: 3, startTime: 4, endTime: 5, drinks: 6, douhan: 7, raiten: 8, remarks: 9, }; // ── 手動実行：今月分のみ ───────────────────────────────────── function updateSalarySheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const now = new Date(); const thisYM = now.getFullYear() * 100 + (now.getMonth() + 1); const rows = getTargetRows(ss, thisYM, thisYM); if (rows.length === 0) { SpreadsheetApp.getUi().alert('今月のデータがありません。'); return; } const groups = groupRows(rows); let count = 0; Object.keys(groups).sort().forEach(key => { const g = groups[key]; const cfg = CONFIG.STAFF[g[0].name]; writeSheet(ss, makeSheetName(cfg.nick, g[0].yearMonth), g, g[0].name, cfg); count++; }); SpreadsheetApp.getUi().alert('完了！' + count + 'シートを更新しました。'); } // ── 先月分 ────────────────────────────────────────────────── function updateLastMonth() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const now = new Date(); let y = now.getFullYear(), m = now.getMonth(); if (m === 0) { y--; m = 12; } const lastYM = y * 100 + m; const rows = getTargetRows(ss, lastYM, lastYM); if (rows.length === 0) { SpreadsheetApp.getUi().alert('先月のデータがありません。'); return; } const groups = groupRows(rows); let count = 0; Object.keys(groups).sort().forEach(key => { const g = groups[key]; const cfg = CONFIG.STAFF[g[0].name]; writeSheet(ss, makeSheetName(cfg.nick, g[0].yearMonth), g, g[0].name, cfg); count++; }); SpreadsheetApp.getUi().alert('先月分 完了！' + count + 'シートを更新しました。'); } // ── フォーム送信トリガー ───────────────────────────────────── function onFormSubmit(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const row = e.values; const name = String(row[COL.name] || '').trim(); if (!CONFIG.STAFF[name]) return; const workDate = parseDate(row[COL.workDate]); if (!workDate) return; const ym = workDate.getFullYear() * 100 + (workDate.getMonth() + 1); const allRows = getTargetRows(ss, ym, ym); const target = allRows.filter(r => r.name === name); if (target.length === 0) return; const cfg = CONFIG.STAFF[name]; writeSheet(ss, makeSheetName(cfg.nick, ym), target, name, cfg); } // ── 対象行取得 ─────────────────────────────────────────────── function getTargetRows(ss, fromYM, toYM) { const sheet = ss.getSheetByName(CONFIG.FORM_SHEET); if (!sheet) return []; const data = sheet.getDataRange().getValues(); const rows = []; for (let i = 1; i < data.length; i++) { const row = data[i]; const name = String(row[COL.name] || '').trim(); if (!CONFIG.STAFF[name]) continue; const workDate = parseDate(row[COL.workDate]); if (!workDate) continue; const ym = workDate.getFullYear() * 100 + (workDate.getMonth() + 1); if (ym < fromYM || ym > toYM) continue; rows.push({ timestamp: row[COL.timestamp], name, workDate, startTime: row[COL.startTime], endTime: row[COL.endTime], drinks: String(row[COL.drinks] || '').trim(), douhan: String(row[COL.douhan] || '').trim(), raiten: String(row[COL.raiten] || '').trim(), remarks: String(row[COL.remarks] || '').trim(), yearMonth: ym, }); } return rows; } function groupRows(rows) { const g = {}; rows.forEach(r => { const key = r.yearMonth + '_' + r.name; if (!g[key]) g[key] = []; g[key].push(r); }); return g; } function makeSheetName(nick, ym) { return nick + String(ym).slice(2,4) + '.' + String(ym).slice(4,6); } // ── シート書き込み ─────────────────────────────────────────── function writeSheet(ss, sheetName, rows, staffName, cfg) { const existing = ss.getSheetByName(sheetName); if (existing) ss.deleteSheet(existing); const ws = ss.insertSheet(sheetName); const isTiered = cfg.type === 'tiered'; // tiered: 14列、flat: 13列 let headers, widths; if (isTiered) { headers = [ '氏名','勤務日','出勤時間','退勤時間','備考','時間', '給料\n(〜22時\n1,300円)', '給料\n(22時〜\n1,500円)', '同伴','シャンパン\n/ワイン','来店\n(セット×15%)','インスタ','手当計','合計' ]; widths = [80,60,60,60,220,45,75,75,55,70,80,55,60,75]; } else { headers = [ '氏名','勤務日','出勤時間','退勤時間','備考','時間', '給料' + cfg.rate + '円', '同伴','シャンパン\n/ワイン','来店\n(セット×15%)','インスタ','手当計','合計' ]; widths = [80,60,60,60,220,45,90,55,70,80,55,60,75]; } const totalCols = headers.length; const hRange = ws.getRange(1, 1, 1, totalCols); hRange.setValues([headers]); hRange.setBackground('#D9E1F2'); hRange.setFontWeight('bold'); hRange.setHorizontalAlignment('center'); hRange.setWrap(true); ws.setRowHeight(1, 52); widths.forEach((w, i) => ws.setColumnWidth(i+1, w)); // データ行 const sorted = rows.slice().sort((a,b) => a.workDate - b.workDate); const dataArr = sorted.map(r => { const mins = calcMinutes(r.startTime, r.endTime); const hours = mins > 0 ? Math.floor(mins / 30) * 0.5 : ''; const dou = r.douhan.trim() !== '' ? CONFIG.ALLOWANCE.douhan : 0; const drk = calcDrinks(r.drinks); const rai = calcRaiten(r.raiten); const ins = (r.remarks.includes('インスタ投稿') || r.remarks.includes('インスタ更新')) ? CONFIG.ALLOWANCE.insta : 0; const allowTotal = dou + drk + rai + ins; if (isTiered) { const w1 = calcWageBefore22(r.startTime, r.endTime); const w2 = calcWageAfter22(r.startTime, r.endTime); return [ r.name, Utilities.formatDate(r.workDate, 'Asia/Tokyo', 'MM/dd'), formatTime(r.startTime), formatTime(r.endTime), r.remarks, hours||'', w1||'', w2||'', dou||'', drk||'', rai||'', ins||'', allowTotal||'', (w1+w2+allowTotal)||'', ]; } else { const wage = calcWageFlat(r.startTime, r.endTime, cfg.rate); return [ r.name, Utilities.formatDate(r.workDate, 'Asia/Tokyo', 'MM/dd'), formatTime(r.startTime), formatTime(r.endTime), r.remarks, hours||'', wage||'', dou||'', drk||'', rai||'', ins||'', allowTotal||'', (wage+allowTotal)||'', ]; } }); if (dataArr.length > 0) ws.getRange(2, 1, dataArr.length, totalCols).setValues(dataArr); const lastData = dataArr.length + 1; const sumRow = lastData + 1; // 合計行 ws.getRange(sumRow, 5).setValue('合計'); for (let c = 6; c <= totalCols; c++) { ws.getRange(sumRow, c).setFormula('=SUM(' + columnLetter(c) + '2:' + columnLetter(c) + lastData + ')'); } ws.getRange(sumRow, 1, 1, totalCols).setBackground('#FCE4D6').setFontWeight('bold'); // 数値フォーマット ws.getRange(2, 6, lastData, 1 ).setNumberFormat('0.0'); ws.getRange(2, 7, lastData, totalCols-6 ).setNumberFormat('#,##0'); ws.getRange(sumRow, 6, 1, 1 ).setNumberFormat('0.0'); ws.getRange(sumRow, 7, 1, totalCols-6 ).setNumberFormat('#,##0'); // 枠線 ws.getRange(1, 1, sumRow, totalCols).setBorder(true,true,true,true,true,true); // 手当列の色（tiered: 9〜12列目、flat: 8〜11列目） const aStart = isTiered ? 9 : 8; ws.getRange(2, aStart, lastData, 4).setBackground('#FFF2CC'); ws.getRange(1, aStart, 1, 4).setBackground('#FFD966').setFontWeight('bold'); } function columnLetter(col) { return col <= 26 ? String.fromCharCode(64 + col) : 'A' + String.fromCharCode(64 + col - 26); } // ── 給料計算 ───────────────────────────────────────────────── function calcWageFlat(startVal, endVal, rate) { const s = parseTimeMin(startVal), e = parseTimeMin(endVal); if (s === null || e === null) return 0; const end = e < s ? e + 1440 : e; const mins = end - s; if (mins <= 0) return 0; return Math.round(Math.floor(mins / 30) * 30 / 60 * rate); } function calcWageBefore22(startVal, endVal) { const s = parseTimeMin(startVal), e = parseTimeMin(endVal); if (s === null || e === null) return 0; const end = e < s ? e + 1440 : e; const t22 = 1320; const se = Math.max(s, 1200); if (se >= end || se >= t22) return 0; const seg = Math.min(end, t22); return Math.round(Math.floor((seg - se) / 30) * 30 / 60 * CONFIG.TIERED.before22); } function calcWageAfter22(startVal, endVal) { const s = parseTimeMin(startVal), e = parseTimeMin(endVal); if (s === null || e === null) return 0; const end = e < s ? e + 1440 : e; const t22 = 1320; const seg2 = Math.max(Math.max(s, 1200), t22); if (seg2 >= end) return 0; return Math.round(Math.floor((end - seg2) / 30) * 30 / 60 * CONFIG.TIERED.after22); } // ── 手当計算 ───────────────────────────────────────────────── function calcDrinks(val) { const n = parseInt(val, 10); return (!isNaN(n) && n > 0) ? n * CONFIG.ALLOWANCE.champagne : 0; } function calcRaiten(val) { if (!val || val.trim() === '') return 0; const m1 = val.match(/セット[料金金額]*\s*[：:￥¥]?\s*(\d[\d,，]*)/); if (m1) return Math.round(parseInt(m1[1].replace(/[,，]/g,''), 10) * CONFIG.ALLOWANCE.raitenRate); const m2 = val.match(/(\d[\d,，]*)\s*円/); if (m2) return Math.round(parseInt(m2[1].replace(/[,，]/g,''), 10) * CONFIG.ALLOWANCE.raitenRate); const m3 = val.match(/(\d+)/); if (m3) return Math.round(parseInt(m3[1], 10) * CONFIG.ALLOWANCE.raitenRate); return 0; } // ── ユーティリティ ─────────────────────────────────────────── function parseTimeMin(val) { if (val === null || val === undefined || val === '') return null; if (typeof val === 'number') { // 数値型：1未満は通常時刻、1以上は24時超え return Math.round(val * 24 * 60); } const str = String(val).replace(/^'+/, '').trim(); if (!str) return null; const m = str.match(/(\d{1,2}):(\d{2})/); if (!m) return null; return parseInt(m[1], 10) * 60 + parseInt(m[2], 10); } function formatTime(val) { if (val === null || val === undefined || val === '') return ''; if (typeof val === 'number') { const t = Math.round(val * 24 * 60) % 1440; return ('0'+Math.floor(t/60)).slice(-2) + ':' + ('0'+(t%60)).slice(-2); } return String(val).replace(/^'+/, '').trim(); } function calcMinutes(s, e) { const sm = parseTimeMin(s), em = parseTimeMin(e); if (sm === null || em === null) return 0; return Math.max(0, (em < sm ? em + 1440 : em) - sm); } function parseDate(val) { if (!val) return null; if (val instanceof Date) return isNaN(val.getTime()) ? null : val; const d = new Date(val); return isNaN(d.getTime()) ? null : d; }
